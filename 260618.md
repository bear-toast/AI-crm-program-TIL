-- 가리킴 받는 쪽(genre)을 먼저 만든다
CREATE TABLE genre (
    id   INT PRIMARY KEY,             -- 장르 번호: 대표 번호(안 겹침·안 빔)
    name VARCHAR(20) NOT NULL UNIQUE  -- 장르 이름: 꼭 채우고(NOT NULL) 안 겹치게(UNIQUE)
);

-- 가리키는 쪽(content)을 나중에 만든다
CREATE TABLE content (
    id           INT PRIMARY KEY AUTO_INCREMENT,                 -- 콘텐츠 번호: 대표 번호 + 자동 증가
    title        VARCHAR(50) NOT NULL,                           -- 제목: 꼭 채워야 함
    type         VARCHAR(10) DEFAULT 'movie',                    -- 종류: 안 적으면 'movie'
    release_year INT,                                            -- 공개 연도: 특별한 규칙 없음
    rating       DECIMAL(3,1) CHECK (rating >= 0 AND rating <= 10), -- 평점: 0~10만 허용
    genre_id     INT,                                            -- 장르 번호
    FOREIGN KEY (genre_id) REFERENCES genre(id)                  -- genre에 실제 있는 번호만 허용
);
